                       --header 'Authorization: ApiKey *********' \
                       --header 'Content-Type: application/json' \
                       --data-raw '{
                          "name": "REPLACE_WITH_name",
                          "payload": {
                              "$user_id": "REPLACE_WITH_$user_id",
                              "$first_name": "REPLACE_WITH_$first_name",
                              "$last_name": "REPLACE_WITH_$last_name",
                              "token": "*********"
                          }
                      }'
